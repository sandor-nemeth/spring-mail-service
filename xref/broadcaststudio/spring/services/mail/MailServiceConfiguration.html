1   package broadcaststudio.spring.services.mail;

3   import org.apache.commons.lang3.StringUtils;
4   import org.springframework.beans.factory.annotation.Autowired;
5   import org.springframework.boot.SpringApplication;
6   import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
7   import org.springframework.context.annotation.Bean;
8   import org.springframework.context.annotation.ComponentScan;
9   import org.springframework.context.annotation.Configuration;
10  import org.springframework.mail.MailSender;
11  import org.springframework.mail.javamail.JavaMailSenderImpl;
12  
13  import broadcaststudio.spring.services.mail.configuration.SMTPConfiguration;
14  
15  /**
16   * The main Spring configuration class of the service.
17   * 
18   * @author sandornemeth
19   * @since 0.0.1
20   */
21  @Configuration
22  @EnableAutoConfiguration
23  @ComponentScan
24  public class MailServiceConfiguration {
25  
26    @Autowired
27    private SMTPConfiguration smtpConfiguration;
28  
29    public static void main(String[] args) {
30      SpringApplication.run(MailServiceConfiguration.class, args);
31    }
32  
33    @Bean
34    public MailSender mailSender() {
35      JavaMailSenderImpl mailSender = new JavaMailSenderImpl();
36      mailSender.setHost(smtpConfiguration.getHost());
37      mailSender.setPort(smtpConfiguration.getPort());
38      mailSender.setProtocol(smtpConfiguration.getProtocol());
39      if (!StringUtils.isBlank(smtpConfiguration.getUsername())) {
40        mailSender.setUsername(smtpConfiguration.getUsername());
41        mailSender.setPassword(smtpConfiguration.getPassword());
42      }
43      return mailSender;
44    }
45  }
